// Reality Check

Before we get into it — PowerSchool doesn't have a public API you can just call from a website. Here's what's actually going on under the hood and what your real options are.

PowerSchool's student-facing data is locked behind your school district's login portal. There's no official API key you can request as a student. Every method below involves some level of workaround.

Your three real options, ranked from easiest to most powerful:

A
Browser Extension easiest
A Chrome/Firefox extension scrapes your PowerSchool page while you're logged in and sends the data to your site. No server required. Works offline. Main downside: you have to be logged into PS in the same browser.
B
Proxy Server recommended
A small Node.js or Python server (runs on your machine or a free host like Railway/Render) logs into PowerSchool on your behalf using your credentials, scrapes or fetches the data, and serves it to your site as JSON. This is what most student-built PS skins actually do.
C
Official API school must enable it
PowerSchool has a real REST API, but it's only accessible if your district's PowerSchool admin enables it and gives you OAuth credentials. Most schools don't give students access. Worth asking your IT department though.

// Option A — Browser Extension

This is the fastest way to get real data showing. The extension runs in your browser, reads your PowerSchool page, and makes the data available to your site via window.postMessage or local storage.

Step 1 — Create the extension

Create a folder called gsq-extension/ with these files:

// gsq-extension/manifest.json
{
  "manifest_version": 3,
  "name": "grades.scotch.quest sync",
  "version": "1.0",
  "permissions": ["storage", "scripting"],
  "content_scripts": [{
    "matches": ["*://YOUR-SCHOOL.powerschool.com/*"],
    "js": ["content.js"]
  }],
  "action": { "popup": "popup.html" }
}
// gsq-extension/content.js — runs on your PS page
// Scrapes grades from the PowerSchool student portal DOM

function scrapeGrades() {
  const courses = [];

  // PowerSchool puts grades in a table — selectors vary by school theme
  // Inspect your PS page and adjust these selectors
  const rows = document.querySelectorAll('table.linkDescList tr');

  rows.forEach(row => {
    const cells = row.querySelectorAll('td');
    if (cells.length < 3) return;

    courses.push({
      name:   cells[0]?.innerText?.trim(),
      grade:  cells[1]?.innerText?.trim(),
      pct:    parseFloat(cells[2]?.innerText) || null,
      period: cells[3]?.innerText?.trim(),
      teacher:cells[4]?.innerText?.trim(),
    });
  });

  return courses.filter(c => c.name);
}

// Save to chrome.storage so popup + your site can read it
const data = { courses: scrapeGrades(), updatedAt: Date.now() };
chrome.storage.local.set({ psData: data });
console.log('[gsq] scraped', data.courses.length, 'courses');

// Option B — Proxy Server (Recommended)

A Node.js server runs on your machine (or free cloud host). It logs into PowerSchool using your credentials, scrapes the grades page, and exposes them as a JSON API that your site calls. This works from any browser and doesn't require an extension.

You're storing your PS password on a server. Only do this on a server you control. Never commit credentials to GitHub — use environment variables.

Step 1 — Set up the proxy server

# In your terminal:
mkdir gsq-proxy && cd gsq-proxy
npm init -y
npm install express axios cheerio cors dotenv
// gsq-proxy/server.js
require('dotenv').config();
const express = require('express');
const axios   = require('axios');
const cheerio = require('cheerio');
const cors    = require('cors');

const app = express();
app.use(cors({ origin: 'https://grades.scotch.quest' }));

const PS_BASE = process.env.PS_URL;   // e.g. https://ps.yourschool.org
const PS_USER = process.env.PS_USER;  // your PS username
const PS_PASS = process.env.PS_PASS;  // your PS password

let sessionCookie = null;

async function psLogin() {
  // Step 1: get the login page to grab the CSRF token
  const loginPage = await axios.get(`${PS_BASE}/public/home.html`);
  const $ = cheerio.load(loginPage.data);
  const pstoken = $('input[name="pstoken"]').val();
  const cookies = loginPage.headers['set-cookie']?.join('; ') || '';

  // Step 2: POST credentials
  const loginRes = await axios.post(
    `${PS_BASE}/guardian/home.html`,
    new URLSearchParams({
      account:   PS_USER,
      pw:        PS_PASS,
      ldappassword: PS_PASS,
      pstoken,
    }).toString(),
    {
      headers: {
        'Content-Type': 'application/x-www-form-urlencoded',
        'Cookie': cookies,
      },
      maxRedirects: 0,
      validateStatus: s => s < 400,
    }
  );

  sessionCookie = loginRes.headers['set-cookie']?.join('; ');
  console.log('[gsq-proxy] logged into PowerSchool');
}

async function fetchGrades() {
  if (!sessionCookie) await psLogin();

  const res = await axios.get(`${PS_BASE}/guardian/home.html`, {
    headers: { Cookie: sessionCookie }
  });

  const $ = cheerio.load(res.data);
  const courses = [];

  // PowerSchool's grades table — inspect YOUR school's PS to confirm selectors
  $('table.linkDescList tbody tr').each((_, row) => {
    const cells = $(row).find('td');
    if (cells.length < 2) return;
    courses.push({
      period:  $(cells[0]).text().trim(),
      name:    $(cells[1]).text().trim(),
      grade:   $(cells[4]).text().trim(),
      pct:     parseFloat($(cells[5]).text()) || null,
      teacher: $(cells[2]).text().trim(),
      room:    $(cells[3]).text().trim(),
    });
  });

  return courses;
}

app.get('/api/grades', async (req, res) => {
  try {
    const grades = await fetchGrades();
    res.json({ ok: true, courses: grades, updatedAt: Date.now() });
  } catch (err) {
    sessionCookie = null; // force re-login next time
    res.status(500).json({ ok: false, error: err.message });
  }
});

app.get('/api/health', (_, res) => res.json({ ok: true }));

app.listen(3001, () => console.log('[gsq-proxy] running on :3001'));

Step 3 — Call it from your site

In shared.js, replace the static D.courses array with a fetch call:

// In shared.js, inside DOMContentLoaded:
async function loadFromProxy() {
  try {
    const res = await fetch('http://localhost:3001/api/grades'); // or your deployed URL
    const json = await res.json();
    if (json.ok && json.courses.length) {
      D.courses = json.courses.map(c => ({
        p:     c.period,
        name:  c.name,
        grade: c.grade || '—',
        pct:   c.pct || 0,
        lc:    (c.grade || 'A')[0],
        tchr:  c.teacher,
        room:  c.room,
        cr:    3.0,
        miss:  0,
        trend: '▶',
        code:  '',
      }));
    }
  } catch(e) {
    console.warn('[gsq] proxy unreachable, using demo data');
  }
}

window.addEventListener('DOMContentLoaded', async () => {
  if (!auth()) return;
  await loadFromProxy();
  // ... rest of your init
});

// Option C — Official PowerSchool API

If your school's IT department enables the PowerSchool API and gives you OAuth2 client credentials, you can use the official REST API. This is the cleanest approach but requires school cooperation.

Ask your school IT admin: "Can I get API access to PowerSchool for a student project?" Reference the PowerSchool Developer Portal at developer.powerschool.com.

If they say yes, you'll get a client_id and client_secret. Then in your proxy server:

// OAuth2 token exchange
const tokenRes = await axios.post(
  `${PS_BASE}/oauth/access_token`,
  'grant_type=client_credentials',
  {
    auth: { username: CLIENT_ID, password: CLIENT_SECRET },
    headers: { 'Content-Type': 'application/x-www-form-urlencoded' }
  }
);
const token = tokenRes.data.access_token;

// Now call the API
const grades = await axios.get(
  `${PS_BASE}/ws/v1/student/{studentId}/section`,
  { headers: { Authorization: `Bearer ${token}` } }
);

// Making Login Actually Authenticate

Right now login.html accepts any username/password and just sets a sessionStorage flag. To make it real:

// In login.html, replace the demo setTimeout block with:
async function doLogin() {
  const u = document.getElementById('uname').value.trim();
  const p = document.getElementById('pw').value;
  if (!u || !p) { showErr(); return; }

  document.getElementById('loading').style.display = 'block';

  try {
    // POST to your proxy server which validates against PowerSchool
    const res = await fetch('https://your-proxy.railway.app/api/login', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ username: u, password: p })
    });
    const json = await res.json();

    if (json.ok) {
      sessionStorage.setItem('gsq_loggedin', '1');
      sessionStorage.setItem('gsq_user', json.username);
      sessionStorage.setItem('gsq_token', json.token); // for API calls
      window.location.href = 'index.html';
    } else {
      showErr();
    }
  } catch (e) {
    showErr();
  }
}
// Add this to your proxy server:
app.post('/api/login', express.json(), async (req, res) => {
  const { username, password } = req.body;
  try {
    PS_USER = username;
    PS_PASS = password;
    sessionCookie = null;
    await psLogin(); // will throw if creds are wrong
    res.json({ ok: true, username, token: Buffer.from(username+':'+Date.now()).toString('base64') });
  } catch(e) {
    res.status(401).json({ ok: false, error: 'Invalid credentials' });
  }
});
